/users get - получение всех пользователей (200 - успешное получение всех пользователей, 400 - неверный запрос, 500 - ошибка сервера)
/users - post - добавление пользователя (200 - успешное добавление пользователя, 400 - неверный запрос, 500 - ошибка сервера)
/users/{id} - get - получение конкретного пользователя (200 - успешное получение пользователя, 400 - неверный запрос, 500 - ошибка сервера)
/users/{id} - patch - изменение конкретного пользователя (200 - успешное изменение пользователя, 400 - неверный запрос, 500 - ошибка сервера)
/users/{id} - delete - удаление конкретного пользователя (200 - успешное удаление пользователя, 400 - неверный запрос, 500 - ошибка сервера)
